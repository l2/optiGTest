0021 function txtFun=boolFun(txtSymbol)
0022 txtFun='';
0023 if isa(txtSymbol,'function_handle')
0024     txtFun=txtSymbol;
0025 else
0026     switch txtSymbol
0027         case {'==','eq'}
0028             txtFun=@eq;
0029         case {'~=','ne'}
0030             txtFun=@ne;
0031         case {'<=','=<','le'}
0032             txtFun=@le;
0033         case {'>=','=>','ge'}
0034             txtFun=@ge;
0035         case {'<','lt'}

0039     end
0040 end

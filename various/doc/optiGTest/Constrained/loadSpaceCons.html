0024 function [xMin,xMax]=loadSpaceCons(funName,dim)
0025 listSpace=struct(...
0026     'RosenbrockCubicLine',[-1.5 -0.5;1.5 2.5],...
0027     'RosenbrockDisk',1.5*[-1 -1;1 1],...
0028     'BirdDisk',[-10 -6.5;0 0],...
0029     'Townsend',[-2.25 -2.5;2.5 1.75],...
0030     'Simionescu',1.25*[-1 -1;1 1]);
0031 
0032 spaceL=listSpace.(funName);
0033 %
0034 if size(spaceL,2)==1&&~isinf(dim)&&~isnan(dim)
0035     xMin=ones(1,dim)*spaceL(1);
0036     xMax=ones(1,dim)*spaceL(2);
0037 else
0038     xMin=spaceL(1,:);
0039     xMax=spaceL(2,:);
0040 end
